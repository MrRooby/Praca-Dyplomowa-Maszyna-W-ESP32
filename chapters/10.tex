\chapter{Spis skrótów i symboli}

\begin{itemize}
  \item[SWIM] system wizualizacji interfejsu Maszyny W
  \item[JAL] jednostka arytmetyczno-logiczna
  \item[RGB] czerwony, zielony, niebieski (ang. \textit{red, green, blue})
  \item[LED] dioda elektroluminescencyjna (ang. \textit{light-emitting diode})
  \item[SMT] montaż powierzchniowy (ang. \textit{surface-mount technology})
  \item[HTTP] protokół przesyłania dokumentów hipertekstowych (ang. \textit{Hypertext Transfer Protocol})
  \item[DNS] system nazw domen (ang. \textit{Domain Name Server})
  \item[TCP] protokół sterowania transmisją (ang. \textit{Transmission Control Protocol})
  \item[PBL] metoda projektów (ang. \textit{Project-Based Learning})
  % \item[JST] standard złączy rastrowych (ang. \textit{Japan Solderless Terminal})
  \item[USB] uniwersalna magistrala szeregowa (ang. \textit{universal serial bus})
  % \item[ARK] listwa zaciskowa śrubowa (niem. \textit{Anschlus Reihenklemme Klemme})
  \item[GND] masa (ang. \textit{ground})
\end{itemize}
